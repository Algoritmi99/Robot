47 // initialize timer 5 to fire interrupt every millisecond at an 8MHz clock
48 void timeTask_init(void) {
49  // prevent multiple initializations
50  if (! timeTask_initialized) {
51  timeTask_initialized = true;
52 
53  // disable power reduction of timer 5
54  PRR1 &= ~_BV(PRTIM5);
55 
56  TCCR5B = 0x00; // stop timer 5 by selecting no clock
57 
58  // set counter value to zero
59  TCNT5 = 0;
60 
61  // set timer 5 to CTC mode (clear timer on compare match with OCR5A)
62  // use prescaler of 8 => timer frequency of 1MHz
63  // TOP = max value of counter is defined by OCR5A
64  // set OCR5A to 1000 = 0x03E8, produces a compare match interrupt frequency of 1kHz = 1ms
65  OCR5A = 0x03E8;
66  TCCR5C = 0x00; // no force output compare
67  TCCR5A = 0x00; // WGM51=0, WGM50=0, COM5A0/1=0, COM5B0/1=0, COM5C0/1=0 (no compare output mode)
68  TIMSK5 = _BV(OCIE5A); // timer 5 output compare A match interrupt enable
69  TCCR5B = _BV(WGM52) | _BV(CS51); // CTC mode (TOP=OCR5A), Clock Select 1 => Prescaler 8, Timer frequency: 1MHz, start timer
70  }
